# Solution 2:
a = "this is the first example"
b = "nothing"
length_a = len(a) - a.count(" ")
length_b = len(b) - b.count(" ")
answer = max(length_a, length_b)
print("The longest string has",answer,"characters in total")

# Solution 3:
